\begin{frame}{Deft in 3 bullets}
\begin{itemize}
  \item Deft is a disaggregated-memory B+Tree tuned for RDMA.
  \item Core ideas: fine-grained node access, scalable locking, lightweight consistency.
  \item My focus: \hl{port communication path to simulated CXL} without rewriting Tree logic.
\end{itemize}
\end{frame}

\begin{frame}{Project Objective}
\begin{itemize}
  \item Keep index semantics stable; swap transport backend.
  \item Build repeatable benchmarking/regression harness.
  \item Stress system and document real failure modes + fixes.
\end{itemize}
\end{frame}

\section{Code Path}

\begin{frame}[fragile]{Compile-Time Backend Switch (CMakeLists.txt)}
\begin{lstlisting}[language=C++]
option(USE_CXL "Build with simulated CXL backend" OFF)

if(USE_CXL)
  add_definitions(-DUSE_CXL)
  set(LINKS_FLAGS ... rt)
else()
  add_definitions(-DUSE_RDMA)
  set(LINKS_FLAGS ... ibverbs ...)
endif()

if(USE_CXL)
  list(FILTER COMMON_FILE EXCLUDE REGEX ".*/rdma/.*\\.cpp$")
else()
  list(FILTER COMMON_FILE EXCLUDE REGEX ".*/cxl/.*\\.cpp$")
endif()
\end{lstlisting}

\footnotesize
This keeps one codebase, but cleanly excludes transport-specific sources.
\end{frame}

\begin{frame}[fragile]{Core CXL Transport API (include/CxlTransport.h)}
\begin{lstlisting}[language=C++]
struct SharedRegion {
  std::string shm_name;
  void *base_addr;
  uint64_t size;
  int fd;
};

SharedRegion create_region(const std::string&, uint64_t);
SharedRegion open_region(const std::string&, uint64_t);

inline void read(void *local, const void *remote, uint64_t size);
inline void write(void *remote, const void *local, uint64_t size);
inline bool cas(void *remote, uint64_t expect, uint64_t val, uint64_t *old);
\end{lstlisting}

\footnotesize
This is the replacement for verbs data-plane operations.
\end{frame}

\begin{frame}[fragile]{Why CXL Multi-Client Hung: Queue Layout Bug}
\small
\begin{block}{Before (conceptual)}
Queue selection used only thread/dir dimensions, effectively colliding across clients.
\end{block}

\begin{block}{After (0a457ef)}
Request/reply queues keyed by client and thread.
\end{block}

\begin{lstlisting}[language=C++]
// src/cxl/CxlTransport.cpp
uint64_t rpc_region_size(uint32_t num_clients,
                         uint32_t num_app_threads,
                         uint32_t num_directories) {
  // req: clients * threads * dirs
  // rep: clients * threads
}
\end{lstlisting}

\footnotesize
Also fixed server/client RPC region sizing and metadata publication.
\end{frame}

\begin{frame}[fragile]{SX-Lock Deadlock Root Cause and Fix (7233322)}
\small
\begin{block}{Problem}
CXL FAA path did plain 64-bit add and ignored masked/boundary semantics needed by lock counters.
\end{block}

\begin{block}{Fix}
Implemented CAS-loop emulation for masked-boundary FAA in CXL path:
\sys{FaaBound(...)} and \sys{FaaDmBound(...)}.
\end{block}

\begin{lstlisting}[language=C++]
// src/dsm_client.cpp (CXL path)
void DSMClient::FaaBound(...) {
  // CAS loop on lanes defined by mask/boundary
  // preserve SX-lock counter semantics
}
\end{lstlisting}
\end{frame}
